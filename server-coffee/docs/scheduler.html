add_daemon = (host, call_back) ->
  daemon.status host,
          (err, res, status) ->
            if err
              call_back err
            else
              daemons_count++
              daemons[host] = res
              call_back null, res

Check if the daemon is alive, maybe we lost the connection.
We want to remove it from the scheduler if it is dead.
             		                            
daemon_is_alive = (host, call_back) ->
  daemon.is_alive host,
          (err, alive, status) ->
            if !alive
              daemons_count--
              daemons[host] = null

Check if the daemon is free
             		                            
daemon_is_free = (host, call_back) ->
  daemon.is_free host,
           (err, is_free, status) ->
            if err
              call_back false
            else
              call_back is_free

remove_daemon = (host) ->
  if daemons[host]
    daemon.stop host, () -> {}
    daemons[host] = null

run_on_free_daemon = (params, call_back) ->
  called = false
  for host in daemons
    if daemons[host]
      daemon_is_free host,
               (freeQ) ->
                if freeQ
                params.host = host
                if !called
                  called = true
                  call_back params
  null
